Anivers 

\starttypescript [sans] [anivers]
  \setups[font:fallback:sans]
  \definefontsynonym [Sans] [Anivers-Regular] [features=default]
\stoptypescript

\starttypescript [sans] [anivers]
  \definefontsynonym [Anivers-Regular] [name:Anivers Regular] [features=default]
\stoptypescript

\starttypescript [anivers]
  \definetypeface [anivers] [ss] [sans] [anivers] [default]
\stoptypescript

Fontin / Fontin Sans 

\definefontfeature [fontin] [method=node, script=latn,
                             liga=yes, %ligatures
                             frac=yes, %fractions
                             pnum=yes, % 
                             tnum=yes] %
 
\starttypescript [sans] [fontin-sans]
  \setups[font:fallback:sans]
  \definefontsynonym [Sans]           [FontinSans-Regular]
  \definefontsynonym [SansBold]       [FontinSans-Bold]
  \definefontsynonym [SansItalic]     [FontinSans-Italic]
  \definefontsynonym [SansBoldItalic] [FontinSans-BoldItalic]
  \definefontsynonym [SansCaps]       [FontinSans-Caps]
\stoptypescript
 
\starttypescript [sans] [fontin-sans]
  \definefontsynonym [FontinSans-Regular]    [name:Fontin Sans Regular]     [features=fontin]
  \definefontsynonym [FontinSans-Bold]       [name:Fontin Sans Bold]        [features=fontin]
  \definefontsynonym [FontinSans-Italic]     [name:Fontin Sans Italic]      [features=fontin]
  \definefontsynonym [FontinSans-BoldItalic] [name:Fontin Sans Bold Italic] [features=fontin]
  \definefontsynonym [FontinSans-Caps]       [name:Fontin Sans Small Caps]  [features=fontin]
\stoptypescript
 
\starttypescript [serif] [fontin]
  \setups[font:fallback:serif]
  \definefontsynonym [Serif]       [Fontin-Regular]
  \definefontsynonym [SerifBold]   [Fontin-Bold]
  \definefontsynonym [SerifItalic] [Fontin-Italic]
  \definefontsynonym [SerifCaps]   [Fontin-Caps]
\stoptypescript
 
\starttypescript [serif] [fontin]
  \definefontsynonym [Fontin-Regular] [name:Fontin Regular]    [features=fontin]
  \definefontsynonym [Fontin-Bold]    [name:Fontin Bold]       [features=fontin]
  \definefontsynonym [Fontin-Italic]  [name:Fontin Italic]     [features=fontin]
  \definefontsynonym [Fontin-Caps]    [name:Fontin SmallCaps]  [features=fontin]
\stoptypescript
 
\starttypescript [fontin]
  \definetypeface [\typescriptone] [rm] [serif] [fontin]      [default]
  \definetypeface [\typescriptone] [ss] [sans]  [fontin-sans] [default]
  \definetypeface [\typescriptone] [tt] [mono]  [modern]      [default] [rscale=1.1]
  \definetypeface [\typescriptone] [mm] [math]  [palatino]    [default]
\stoptypescript

  \definefontsynonym [SansItalic] [MuseoSans-500-Italic] [features=default]

\starttypescript [serif] [museo]

\starttypescript [sans] [museo]

  \definefontsynonym [MuseoSans-500-Italic] [name:Museo Sans 500 Italic] [features=default]
\stoptypescript

\starttypescript [museo]
