\usemodule[database]

\defineseparatedlist
  [MyTable]
  [separator=tab,
   left=\bTD,right=\eTD,
   first=\bTR,last=\eTR,
   before=\bTABLE,after=\eTABLE]

\setupTABLE[c][each][align=flushright,width=2.5cm,align=middle]
