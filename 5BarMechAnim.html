\startJSpreamble stepper used now
  var state = 0 ;
  var step = 0 ;
  function step_clock ()
    { try
        {
          Walk_Field("clock") ;
          this.dirty = false }
      catch (e)
        { } }
  function start_clock ()
    { try
        { if (state==0)
            { step = app.setInterval ("step_clock()", 20) ;
              step.count = 0 ;
              state = 1 }
          else if (state==1)
            { app.clearInterval (step) ;
              state = 2 }
          else if (state==2)
            { app.clearInterval (step) ;
              Reset_Fields("clock") ;
              Set_Field("clock", "1") ;
              state = 0 } }
      catch (e)
        { } }
  function stop_clock ()
    { try
        { app.clearInterval (step) }
      catch (e)
        { } }
\stopJSpreamble
